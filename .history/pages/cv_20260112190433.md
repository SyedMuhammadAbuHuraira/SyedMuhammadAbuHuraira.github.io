<script>
  // Cosmic Dots Animation
  document.addEventListener('DOMContentLoaded', function() {
    const cosmicDots = document.querySelector('.cosmic-dots');
    
    // Create dots
    for (let i = 0; i < 50; i++) {
      const dot = document.createElement('div');
      dot.className = 'cosmic-dot';
      
      // Random position
      dot.style.left = `${Math.random() * 100}%`;
      
      // Random animation delay and duration
      const delay = Math.random() * 15;
      const duration = 15 + Math.random() * 10;
      dot.style.animationDelay = `${delay}s`;
      dot.style.animationDuration = `${duration}s`;
      
      // Random size
      const size = 0.5 + Math.random() * 1.5;
      dot.style.width = `${size}px`;
      dot.style.height = `${size}px`;
      
      cosmicDots.appendChild(dot);
    }

    // Mouse move effect for dots
    document.addEventListener('mousemove', function(e) {
      const dots = document.querySelectorAll('.cosmic-dot');
      const mouseX = e.clientX / window.innerWidth;
      const mouseY = e.clientY / window.innerHeight;
      
      dots.forEach(dot => {
        const rect = dot.getBoundingClientRect();
        const dotX = rect.left + rect.width / 2;
        const dotY = rect.top + rect.height / 2;
        
        const dx = e.clientX - dotX;
        const dy = e.clientY - dotY;
        const distance = Math.sqrt(dx * dx + dy * dy);
        
        if (distance < 100) {
          // Create attraction effect
          const force = (100 - distance) / 100;
          const angle = Math.atan2(dy, dx);
          
          dot.style.transform = `translate(${Math.cos(angle) * force * 10}px, ${Math.sin(angle) * force * 10}px)`;
          
          // Increase glow when cursor is near
          dot.style.boxShadow = 
            `0 0 ${4 + force * 8}px #4A90E2,
             0 0 ${8 + force * 12}px #4A90E2,
             0 0 ${12 + force * 16}px #4A90E2`;
        } else {
          dot.style.transform = '';
          dot.style.boxShadow = 
            `0 0 4px #4A90E2,
             0 0 8px #4A90E2,
             0 0 12px #4A90E2`;
        }
      });
    });
  });
</script>
